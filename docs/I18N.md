# Internationalization (i18n) — Coffee Five

## Approach

Spanish-first, English as a toggle. No routing-based i18n (no `/es/` or `/en/` prefixes). Language is stored in a cookie and toggled client-side.

## Language Provider Pattern

```typescript
// src/lib/i18n-context.tsx
'use client';

import { createContext, useContext, useState } from 'react';
import { translations, Locale, TranslationKey } from '@/config/i18n';

const I18nContext = createContext<{
  locale: Locale;
  setLocale: (locale: Locale) => void;
  t: (key: TranslationKey) => string;
}>({
  locale: 'es',
  setLocale: () => {},
  t: (key) => key,
});

export function I18nProvider({ children }: { children: React.ReactNode }) {
  const [locale, setLocale] = useState<Locale>('es');

  const t = (key: TranslationKey): string => {
    return translations[key]?.[locale] || key;
  };

  return (
    <I18nContext.Provider value={{ locale, setLocale, t }}>
      {children}
    </I18nContext.Provider>
  );
}

export const useI18n = () => useContext(I18nContext);
```

## Usage in Components

```tsx
'use client';
import { useI18n } from '@/lib/i18n-context';

export function LanguageToggle() {
  const { locale, setLocale } = useI18n();
  return (
    <button onClick={() => setLocale(locale === 'es' ? 'en' : 'es')}>
      {locale === 'es' ? 'EN' : 'ES'}
    </button>
  );
}

export function SomeComponent() {
  const { t } = useI18n();
  return <h2>{t('section.process')}</h2>;
}
```

## Notes
- Product titles/descriptions come from Shopify (Juan writes them in Spanish; English can be handled via Shopify's translation features or Translate & Adapt app)
- Blog content from Shopify is single-language — Spanish first. English blog translation is a future Sanity feature.
- All hardcoded UI strings must go through the `t()` function.
